34 class M5Base(BaseDecisionTree):
35 """
36 M5Base. Implements base routines for generating M5 PredictionModel trees and rules.
37  
38 The original algorithm M5 was invented by Quinlan:
39  
40 - Quinlan J. R. (1992). Learning with continuous classes. Proceedings of the
41 Australian Joint Conference on Artificial Intelligence. 343--348. World
42 Scientific, Singapore.
43  
44 Yong Wang and Ian Witten made improvements and created M5':
45  
46 - Wang, Y and Witten, I. H. (1997). Induction of model trees for predicting
47 continuous classes. Proceedings of the poster papers of the European
48 Conference on Machine Learning. University of Economics, Faculty of
49 Informatics and Statistics, Prague.
50  
51 Pruning and Smoothing can be activated and deactivated on top of the base
52 model. TODO check if 'rules' should be supported too
53  
54 Inspired by Weka (https://github.com/bnjmn/weka) M5Base class, from Mark Hall
55  
56 Attributes
57 ----------
58 criterion : str, default="friedman_mse"
59 M5 suggests to use the standard deviation (RMSE impurity) instead of
60 variance (MSE impurity) or absolute deviation (MAE impurity) as in CART.
61 According to M5' paper, both 3 criterions are equivalent.
62 splitter : default="best"
63 M5 suggests to take the feature leading to the best gain in criterion.
64 max_depth : default None
65 This is not used in the original M5 article, hence default=None.
66 min_samples_split : int, default=4
67 M5 suggests a value of 4 so that each leaf will have at least 2 samples.
68 min_samples_leaf : int, default=2
69 M5' suggest to add this explicitly to avoid zero-variance in leaves, and
70 n<=p for constant models.
71 min_weight_fraction_leaf : float, default=0.0
72 This is not used in the original M5 article, hence default=0.0.
73 TODO this would actually maybe be better than min_sample_leaf ?
74 max_features : default None
75 This is not used in the original M5 article, hence default is None.
76 max_leaf_nodes : int, default None
77 This is not used in the original M5 article, hence default is None.
78 min_impurity_decrease : float, default=0.0
79 This is not used in the original M5 article, hence default is None.
80 class_weight : default None
81 This is not used (?)
82 leaf_model : RegressorMixin
83 The regression model used in the leaves. This instance will be cloned
84 for each leaf.
85 use_pruning : bool
86 If False, pruning will be disabled.
87 use_smoothing : bool or str {'installed', 'on_prediction'}, default None
88 None and True means 'installed' by default except if smoothing_constant
89 or smoothing_constant_ratio is 0.0
90 smoothing_constant: float, default None
91 The smoothing constant k defined in the original M5 article, used as the
92 weight for each parent model in the recursive weighting process.
93 If None, the default value from the paper (k=15) will be used.
94 smoothing_constant_ratio: float, default None
95 An alternate way to define the smoothing constant, as a ratio of the
96 total number of training samples. The resulting smoothing constant will
97 be smoothing_constant_ratio * n where n is the number of samples.
98 Note that this may not be an integer.
99 debug_prints: bool, default False
100 A boolean to enable debug prints
101 ccp_alpha: float, default 0.0
102 TODO is this relevant ? does that conflict with "use_pruning" ?
103 random_state : None, int, or RandomState, default=None
104 See `RegressionTree.random_state`.
105 """

185 """Fit a M5Prime model.
186  
187 Parameters
188 ----------
189 X : numpy.ndarray
190 y : numpy.ndarray
191 sample_weight
192 check_input
193 X_idx_sorted
194  
195 Returns
196 -------
197  
198 """
199 # (0) smoothing default values behaviour
200 if self.smoothing_constant_ratio == 0.0 or self.smoothing_constant == 0:
201 self.use_smoothing = False
202 elif self.use_smoothing is None or self.use_smoothing is True:
203 # default behaviour
204 if isinstance(self.leaf_model, LinearRegression):
205 self.use_smoothing = "installed"
206 else:
207 self.use_smoothing = "on_prediction"
208

210 if self.use_smoothing not in [False, np.bool_(False), "installed", "on_prediction"]:
211 raise ValueError("use_smoothing: Unexpected value: %s, please report it as issue." % self.use_smoothing)

214 super(M5Base, self).fit(X, y, sample_weight=sample_weight, check_input=check_input, X_idx_sorted=X_idx_sorted)

319 """
320 Reduces the size of this object by removing from internal structures

358 def _compress(old_node_id):
359 """
360  
361 Parameters
362 ----------
363 old_node_id
364  
365 Returns
366 -------
367 the depth and new indices of left and right children
368  
369 """
370 global next_free_id
371 new_node_id = next_free_id
372 next_free_id += 1
373  
374 # use the old tree to walk
375 old_node = dct["nodes"][old_node_id]
376 left_id = old_node["left_child"]
377 right_id = old_node["right_child"]
378  
379 # Create the new node with a copy of the old
380 new_nodes[new_node_id] = old_node # this is an entire row so it is probably copied already by doing so.
381 new_values[new_node_id] = dct["values"][old_node_id]
382 new_node_models[new_node_id] = copy(old_node_modls[old_node_id])
383  
384 if left_id == _tree.TREE_LEAF:
385 # ... and right_id == _tree.TREE_LEAF
386 # => node_id is a leaf. Nothing to do
387 return 1, new_node_id
388 else:
389  
390 left_depth, new_id_left = _compress(left_id)
391 right_depth, new_id_right = _compress(right_id)
392  
393 # store the new indices
394 new_nodes[new_node_id]["left_child"] = new_id_left
395 new_nodes[new_node_id]["right_child"] = new_id_right
396  
397 return 1 + max(left_depth, right_depth), new_node_id
398  
399 # complete definition of the new tree
400 dct["max_depth"] = _compress(0)[0] - 1 # root node has depth 0, not 1
401 dct["node_count"] = new_node_count # new_nodes.shape[0]
402 dct["nodes"] = new_nodes
403 dct["values"] = new_values
404 new_tree.__setstate__(dct)
405

407 # See https://github.com/scikit-learn/scikit-learn/blob/0.17.1/sklearn/tree/_tree.pyx#L623
408 new_tree.max_depth = dct["max_depth"]
409  
410 # update self fields
411 self.tree_ = new_tree
412 self.node_models = new_node_models

414 def install_smoothing(self, X_train_all, y_train_all, nodes_to_samples, smoothing_constant):
415 """
416 Executes the smoothing procedure described in the M5 and M5P paper,
417 "once for all". This means that all models are modified so that after
418 this method has completed, each model in the tree is already a smoothed
419 model.
420  
421 This has pros (prediction speed) and cons (the model equations are
422 harder to read - lots of redundancy)
423  
424 It can only be done if leaf models are instances of `LinearRegression`
425 """
426 # check validity: leaf models have to support pre-computed smoothing
427 if not isinstance(self.leaf_model, LinearRegression):
428 raise TypeError(
429 "`install_smoothing` is only available if leaf "
430 "models are instances of `LinearRegression` or a "
431 "subclass"
432 )
433  
434 # Select the Error metric to compute model errors (used to compare
435 # node with subtree for pruning)
436 # TODO in Weka they use RMSE, but in papers they use MAE. this could be a parameter.
437 # TODO Shall we go further and store the residuals, or a bunch of metrics? not sure
438 err_metric = root_mean_squared_error # mean_absolute_error
439  
440 # --- Node info
441 # TODO we should probably not do this once in each method, but once or give access directly (no copy)
442 # see https://github.com/scikit-learn/scikit-learn/blob/master/sklearn/tree/_tree.pyx#L637
443 dct = self.tree_.__getstate__().copy()
444 old_node_models = self.node_models
445  
446 m_shape = old_node_models.shape
447 new_node_models = _empty_contig_ar(m_shape, dtype=old_node_models.dtype)

449 def smooth__(
450 coefs_,
451 n_samples,
452 features=None,
453 parent=None, # type: _SmoothingDetails
454 parent_features=None,
455 k=smoothing_constant, #
456 ):
457 # type: (...) -> _SmoothingDetails
458 """
459 Smoothes the model coefficients or intercept `coefs_` (an array or
460 a scalar), using the smoothing results at parent node.
461  
462 At each node we keep in memory three results A, B, C.
463 - the new coef at each node is A + B
464 - the recursion equations are
465 - B(n) = (k / (n_samples(n) + k)) * A(n-1) + B(n-1)
466 - C(n) = (n_samples(n) / (n_samples(n) + k)) * C(n-1)
467 - A(n) = coef(n)*C(n)
468 """
469 if parent is None:
470 # A0 = coef(0), C0 = 1, B0 = 0
471 if features is None:
472 # single scalar value
473 return _SmoothingDetails(A=coefs_, B=0, C=1)
474 else:
475 # vector
476 if len(coefs_) == 0:
477 coefs_ = np.asarray(coefs_)
478 if coefs_.shape[0] != len(features):
479 raise ValueError("nb features does not match the nb " "of coefficients")
480 return _SmoothingDetails(
481 A=coefs_,
482 B=np.zeros(coefs_.shape, dtype=coefs_.dtype),
483 C=np.ones(coefs_.shape, dtype=coefs_.dtype),
484 )
485 else:
486 # B(n) = k/(n_samples(n)+k)*A(n-1) + B(n-1)
487 # C(n) = (n_samples(n) / (n_samples(n) + k)) * C(n-1)
488 Bn = (k / (n_samples + k)) * parent.A + parent.B
489 Cn = (n_samples / (n_samples + k)) * parent.C
490  
491 # A(n) = coef(n) * C(n)
492 if features is None:
493 # single scalar value: easy
494 An = coefs_ * Cn
495 return _SmoothingDetails(A=An, B=Bn, C=Cn)
496 else:
497 # vector of coefs: we have to 'expand' the coefs array
498 # because the coefs at this node apply for (features)
499 # while coefs at parent node apply for (parent_features)
500 An = np.zeros(Cn.shape, dtype=Cn.dtype)
501 parent_features = np.array(parent_features)
502 features = np.array(features)
503  
504 # Thanks https://stackoverflow.com/a/8251757/7262247 !
505 index = np.argsort(parent_features)
506 sorted_parents = parent_features[index]
507 sorted_index = np.searchsorted(sorted_parents, features)
508  
509 features_index = np.take(index, sorted_index, mode="clip")
510 if np.any(parent_features[features_index] != features):
511 # result = np.ma.array(features_index, mask=mask)
512 raise ValueError(
513 "Internal error - please report this."
514 "One feature was found in the child "
515 "node, that was not in the parent "
516 "node."
517 )
518  
519 if len(features_index) > 0:
520 An[features_index] = coefs_ * Cn[features_index]
521  
522 return _SmoothingDetails(A=An, B=Bn, C=Cn)

524 def _smooth(
525 node_id,
526 parent_features=None,
527 parent_coefs: _SmoothingDetails = None,
528 parent_intercept: _SmoothingDetails = None,
529 ):
530 # Gather all info on this node
531 # --base regression tree
532 node_info = dct["nodes"][node_id]
533 left_id = node_info["left_child"]
534 right_id = node_info["right_child"]
535 # --additional model
536 node_model = old_node_models[node_id]
537 # --samples
538 samples_at_this_node = get_samples_at_node(node_id, nodes_to_samples)
539 n_samples_at_this_node = samples_at_this_node.shape[0]
540  
541 # Note: should be equal to tree.n_node_samples[node_id]
542 if n_samples_at_this_node != self.tree_.n_node_samples[node_id]:
543 raise ValueError("n_samples_at_this_node: Unexpected value, please report it as issue.")
544  
545 y_true_this_node = y_train_all[samples_at_this_node]
546 X_this_node = X_train_all[samples_at_this_node, :]
547  
548 # (1) smooth current node
549 parent_features = parent_features if parent_features is not None else None
550 is_constant_leaf = False
551 if left_id == _tree.TREE_LEAF and isinstance(node_model, ConstantLeafModel):
552 is_constant_leaf = True
553 node_features = ()
554 smoothed_features = parent_features if parent_features is not None else node_features
555 node_coefs = ()
556 node_intercept = dct["values"][node_id]
557  
558 # Extract the unique scalar value
559 if len(node_intercept) != 1:
560 raise ValueError("node_intercept: Unexpected value: , please report it as issue." % node_intercept)
561  
562 node_intercept = node_intercept.item()
563  
564 else:
565 # A leaf LinRegLeafModel or a split SplitNodeModel
566 node_features = node_model.features
567 smoothed_features = parent_features if parent_features is not None else node_features
568 node_coefs = node_model.model.coef_
569 node_intercept = node_model.model.intercept_
570  
571 # Create a new linear regression model with smoothed coefficients
572 smoothed_sklearn_model = clone(self.leaf_model)
573 smoothed_coefs = smooth__(
574 node_coefs,
575 features=node_features,
576 n_samples=n_samples_at_this_node,
577 parent=parent_coefs,
578 parent_features=parent_features,
579 )
580 smoothed_intercept = smooth__(node_intercept, n_samples=n_samples_at_this_node, parent=parent_intercept)
581 smoothed_sklearn_model.coef_ = smoothed_coefs.A + smoothed_coefs.B
582 smoothed_sklearn_model.intercept_ = smoothed_intercept.A + smoothed_intercept.B
583  
584 # Finally update the node
585 if is_constant_leaf:
586 smoothed_node_model = LinRegLeafModel(smoothed_features, smoothed_sklearn_model, None)
587 else:
588 smoothed_node_model = copy(node_model)
589 smoothed_node_model.features = smoothed_features
590 smoothed_node_model.model = smoothed_sklearn_model
591  
592 # Remember the new smoothed model
593 new_node_models[node_id] = smoothed_node_model
594  
595 if left_id == _tree.TREE_LEAF:
596 # If this is a leaf, update the prediction error on X
597 y_pred_this_node = smoothed_node_model.predict(X_this_node)
598 smoothed_node_model.error = err_metric(y_true_this_node, y_pred_this_node)
599  
600 else:
601 # If this is a split node - recurse on each subtree
602 _smooth(
603 left_id,
604 parent_features=smoothed_features,
605 parent_coefs=smoothed_coefs,
606 parent_intercept=smoothed_intercept,
607 )
608 _smooth(
609 right_id,
610 parent_features=smoothed_features,
611 parent_coefs=smoothed_coefs,
612 parent_intercept=smoothed_intercept,
613 )
614

681 def _denormalize(node_id, x_scaler, y_scaler):
682 """

709 new_model = copy(old_node_models[node_id])

711  
712 # Create the new value by de-scaling y
713 if y_scaler is not None:

715 # the value will never be used. However to preserve consistency
716 # of the whole values structure and for debugging purposes, we
717 # choose this safe path of denormalizing ALL.
718 # TODO we could also do it at once outside of the recursive
719 # calls, but we should check for side effects
720 new_values[node_id] = y_scaler.inverse_transform(
721 dct["values"][node_id]
722 )
723 else:
724 # no denormalization: simply copy
725 new_values[node_id] = dct["values"][node_id]

729 # => node_id is a leaf

732 return

745 # denormalize the split value if needed
746 if x_scaler is not None:

749 # even if we only want to denormalize one.

759 if isinstance(new_model, SplitNodeModel):

765 "should be SplitNodeModel")

767 return
768  
769 # denormalize the whole tree and put the result in (new_nodes,
770 # new_values, new_node_models) recursively

772  
773 # complete definition of the new tree
774 dct["nodes"] = new_nodes
775 dct["values"] = new_values

778 # update the self fields
779 # self.features_usage
780 self.tree_ = new_tree
781 self.node_models = new_node_models

1307 # already
1308 y_pred_children = predict_from_leaves_no_smoothing(tree, node_models, X_this_node)

1312 # weigthed sum of adjusted errors. Why?
1313 # the total number of parameters if we do not prune, is the sum of
1314 # params in each branch PLUS 1 for the split
1315 n_params_splitmodel = node_models[left_node].n_params + node_models[right_node].n_params + 1
1316 adjusted_err_children = compute_adjusted_error(err_children, n_samples_at_this_node, n_params_splitmodel)
1317  
1318 # (6) compare and either prune at this node or keep the subtrees
1319 std_dev_this_node = np.nanstd(y_true_this_node)
1320 # note: the first criterion is now already checked before that
1321 # function call, in `prune_on_min_impurity`
1322 if use_pruning and (
1323 # TODO these parameters should be in the constructor
1324 # see also 2 comments about min_impurity_split_as_initial_ratio
1325 std_dev_this_node < (global_std_dev * 0.05)
1326 or (adjusted_err_model <= adjusted_err_children)
1327 or (adjusted_err_model < (global_std_dev * 0.00001)) # so this means very good R² already
1328 ):
1329 # Choose model for this node rather than subtree model
1330 # -- prune from this node on
1331 removed_nodes = prune_children(node_id, tree)
1332 node_models[removed_nodes] = _tree.TREE_UNDEFINED
1333  
1334 # store the model information
1335 node_models[node_id] = model_this_node
1336  
1337 # update and return the features used
1338 selected_features_dct = {k: n_samples_at_this_node for k in selected_features_dct.keys()}
1339 return selected_features_dct
1340  
1341 else:
1342 # The subtrees are good or we do not want pruning: keep them.
1343 # This node will remain a split, and will only contain the digest
1344 # about the subtree
1345 node_models[node_id] = SplitNodeModel(
1346 n_params_splitmodel, err_children, selected_features, skmodel_this_node
1347 )
1348  
1349 # return the features used
1350 return selected_features_dct

1353 def compute_adjusted_error(err, n_samples, n_parameters, multiplier=PRUNING_MULTIPLIER):
1354 """
1355 Return the penalized error obtained from `err`, the prediction error at a
1356 given tree node, by multiplying it by ``(n+p)/(n-p)``. `n` is the number of
1357 samples at this node and `p` is the number of parameters of the model.
1358  
1359 According to the original M5 paper, this is to penalize complex models
1360 used for few samples.
1361  
1362 Note that if ``n_samples <= n_parameters`` the denominator is zero or negative.
1363 In this case an arbitrary high penalization factor is used: ``10 * err``.
1364  
1365 Parameters
1366 ----------
1367 err : float
1368 The model error for a given tree node. Typically the MAE.
1369  
1370 n_samples : int
1371 The number of samples at this node.
1372  
1373 n_parameters : int
1374 The number of parameters of the model at this node.
1375  
1376 Returns
1377 -------
1378 penalized_mae : float
1379 The penalized error ``err * (n+p)/(n-p)``, or ``10 * err`` if ``n <= p``.
1380 """
1381 #
1382 if n_samples <= n_parameters:
1383 # denominator is zero or negative: use a large factor so as to penalize
1384 # a lot this overly complex model.
1385 factor = 10.0 # Caution says Yong in his code
1386 else:
1387 factor = (n_samples + multiplier * n_parameters) / (n_samples - n_parameters)
1388  
1389 return err * factor

1518 node_nb_train_samples + smoothing_constant

1577 """
1578 Prunes the children of node_id in the given `tree`.
1579  
1580 Inspired by https://github.com/shenwanxiang/sklearn-post-prune-tree/blob/master/tree_prune.py#L122
1581

1583 are modificable (and are the only things we need to modify to fix the
1584 tree). It seems to be the case as of now.
1585  
1586 Parameters
1587 ----------
1588 node_id : int
1589 tree : Tree
1590  
1591 Returns
1592 -------
1593 removed_nodes : List[int]
1594 A list of removed nodes
1595 """

1597 def _prune_below(_node_id):
1598 left_child = tree.children_left[_node_id]
1599 right_child = tree.children_right[_node_id]
1600 if left_child == _tree.TREE_LEAF:
1601 # _node_id is a leaf: left_ & right_child say "leaf".Nothing to do
1602 return []
1603 else:
1604 # Make sure everything is pruned below: children should be leaves
1605 removed_l = _prune_below(left_child)
1606 removed_r = _prune_below(right_child)
1607  
1608 # -- First declare that they are not leaves anymore but they do not exist at all
1609 for child in [left_child, right_child]:
1610

1612 raise ValueError("Unexpected children_left: %s, please report it as issue." % child)
1613

1618 tree.children_right[child] = _tree.TREE_UNDEFINED
1619  
1620 # -- Then declare that current node is a leaf
1621 tree.children_left[_node_id] = _tree.TREE_LEAF
1622 tree.children_right[_node_id] = _tree.TREE_LEAF
1623  
1624 # Return the list of removed nodes
1625 return removed_l + removed_r + [left_child, right_child]
